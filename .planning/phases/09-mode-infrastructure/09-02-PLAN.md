    **Create src/ui/components/ModePicker.tsx:**

    ```tsx
    import { h } from 'preact'
    import type { GameMode } from '../../machine/gameMachine'

    interface Props {
      mode: GameMode
      onModeChange: (mode: GameMode) => void
    }

    const MODES: { value: GameMode; label: string }[] = [
      { value: 'endless',    label: 'Endless' },
      { value: 'timeAttack', label: 'Time-Attack' },
      { value: 'daily',      label: 'Daily' },
    ]

    export function ModePicker({ mode, onModeChange }: Props) {
      return h(
        'div',
        { className: 'mode-picker', role: 'group', 'aria-label': 'Game mode' },
        ...MODES.map(({ value, label }) =>
          h(
            'button',
            {
              key: value,
              type: 'button' as const,
              className: 'mode-btn' + (mode === value ? ' active' : ''),
              onClick: (e: MouseEvent) => {
                e.stopPropagation()
                onModeChange(value)
              },
              'aria-pressed': mode === value,
            },
            label,
          )
        ),
      )
    }
    ```

    Notes:
    - Uses `h('button', ...)` directly (not the Button wrapper) so we can override styling cleanly per the segmented control pattern without fighting the base button gradient.
    - `e.stopPropagation()` prevents the TitleScreen's onClick from firing START when a mode button is tapped (same pattern used by the settings button in TitleScreen).
    - `aria-pressed` communicates selection state to screen readers (A11Y pattern established in Phase 4).
    - All three modes are always enabled (D-14 — no unlock gating).
