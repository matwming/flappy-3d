```typescript
import {
  DataTexture,
  MeshToonMaterial,
  NearestFilter,
  RGBAFormat,
  UnsignedByteType,
} from 'three'

// Single shared gradient texture — created ONCE, passed to all toon materials.
// 5-band gradient from dark shadow → mid shadow → mid → highlight → specular.
// Width=5 height=1 pixel DataTexture; NearestFilter for hard cel-shaded bands.
export function createToonGradient(): DataTexture {
  const data = new Uint8Array([
    // R    G    B    A    (5 pixels, each representing a shade band)
    // darkest shadow
    40,  40,  40,  255,
    // mid shadow
    100, 100, 100, 255,
    // mid tone
    180, 180, 180, 255,
    // highlight
    230, 230, 230, 255,
    // near-white specular
    255, 255, 255, 255,
  ])
  const texture = new DataTexture(data, 5, 1, RGBAFormat, UnsignedByteType)
  texture.magFilter = NearestFilter  // hard bands, not interpolated
  texture.needsUpdate = true
  return texture
}

// Create a MeshToonMaterial with the shared gradient and a given hex color tint.
// All pipe instances use the same gradient; color differs by tinting material.color.
export function createToonMaterial(
  gradient: DataTexture,
  color: number,
): MeshToonMaterial {
  return new MeshToonMaterial({
    color,
    gradientMap: gradient,
  })
}
```

```typescript
import { WebGLRenderer, Scene, PerspectiveCamera, Vector2 } from 'three'
import { EffectComposer } from 'three/examples/jsm/postprocessing/EffectComposer.js'
import { RenderPass } from 'three/examples/jsm/postprocessing/RenderPass.js'
import { UnrealBloomPass } from 'three/examples/jsm/postprocessing/UnrealBloomPass.js'
import { ShaderPass } from 'three/examples/jsm/postprocessing/ShaderPass.js'
import { VignetteShader } from 'three/examples/jsm/shaders/VignetteShader.js'
import { OutputPass } from 'three/examples/jsm/postprocessing/OutputPass.js'
import { BLOOM_STRENGTH, BLOOM_RADIUS, BLOOM_THRESHOLD, VIGNETTE_OFFSET, VIGNETTE_DARKNESS } from '../constants'

export interface ComposerResult {
  composer: EffectComposer
  render: () => void   // call this instead of renderer.render() each frame
}

export function createComposer(
  renderer: WebGLRenderer,
  scene: Scene,
  camera: PerspectiveCamera,
): ComposerResult | null {
  // Mobile gate (per D-18):
  const isLowTier =
    navigator.hardwareConcurrency <= 4 ||
    /Mobile|Android/i.test(navigator.userAgent)

  if (isLowTier) {
    return null  // caller falls back to renderer.render(scene, camera)
  }

  const w = renderer.domElement.clientWidth
  const h = renderer.domElement.clientHeight

  const composer = new EffectComposer(renderer)
  composer.addPass(new RenderPass(scene, camera))
  composer.addPass(new UnrealBloomPass(new Vector2(w, h), BLOOM_STRENGTH, BLOOM_RADIUS, BLOOM_THRESHOLD))

  const vignettePass = new ShaderPass(VignetteShader)
  vignettePass.uniforms['offset'].value = VIGNETTE_OFFSET
  vignettePass.uniforms['darkness'].value = VIGNETTE_DARKNESS
  composer.addPass(vignettePass)

  composer.addPass(new OutputPass())

  // Resize handler: add to window resize event.
  // IMPORTANT: Do NOT rebuild the composer on resize — just setSize (per D-21 anti-pattern note in CONTEXT).
  window.addEventListener('resize', () => {
    const nw = window.innerWidth
    const nh = window.innerHeight
    composer.setSize(nw, nh)
  })

  return {
    composer,
    render: () => composer.render(),
  }
}
```

NOTE on VignetteShader uniform names: verify `'offset'` and `'darkness'` are correct for the installed Three.js version by checking `node_modules/three/examples/jsm/shaders/VignetteShader.js` uniform names before hardcoding. If names differ, use the actual names from the installed file.

NOTE on OutputPass: verify import path. In some Three.js versions it is `three/examples/jsm/postprocessing/OutputPass.js`. If missing, use `GammaCorrectionShader` with `ShaderPass` as fallback.
